28  import org.springframework.http.ResponseEntity;
29  import org.springframework.web.bind.annotation.PathVariable;

35  /**
36   * The domain group management api.
37   *
38   * @author Christian Bremer
39   */
40  @Api(value = "DomainGroupManagement")
41  public interface DomainGroupManagementApi {
42  
43    /**
44     * Get domain groups.
45     *
46     * @param sort the sort order
47     * @return the groups
48     */
49    @ApiOperation(
50        value = "Get all domain groups.",
51        nickname = "getGroups",
52        notes = "Get all domain groups.",
53        response = DomainGroup.class,
54        responseContainer = "List",
55        tags = {"domain-group-management-controller"})
56    @ApiResponses(value = {
57        @ApiResponse(code = 200, message = "A list of domain groups.",
58            response = DomainGroup.class, responseContainer = "List"),
59        @ApiResponse(code = 500, message = "Fatal server error.",
60            response = RestApiException.class)
61    })
62    @RequestMapping(
63        value = "/api/groups",
64        produces = {"application/json"},
65        method = RequestMethod.GET)
66    ResponseEntity<List<DomainGroup>> getGroups(
67        @ApiParam(value = "The sort order.", defaultValue = DomainGroup.DEFAULT_SORT_ORDER)
68        @RequestParam(value = "sort",
69            defaultValue = DomainGroup.DEFAULT_SORT_ORDER) String sort);
70  
71    /**
72     * Add domain group.
73     *
74     * @param group the domain group to add
75     * @return the added domain group
76     */
77    @ApiOperation(
78        value = "Add domain group.",
79        nickname = "addGroup",
80        response = DomainGroup.class,
81        tags = {"domain-group-management-controller"})
82    @ApiResponses(value = {
83        @ApiResponse(code = 200, message = "The added domain group.", response = DomainGroup.class),
84        @ApiResponse(code = 400, message = "Bad request.",
85            response = RestApiException.class),
86        @ApiResponse(code = 406, message = "Already exists",
87            response = RestApiException.class),
88        @ApiResponse(code = 500, message = "Fatal server error.",
89            response = RestApiException.class)
90    })
91    @RequestMapping(
92        value = "/api/groups",
93        produces = {"application/json"},
94        consumes = {"application/json"},
95        method = RequestMethod.POST)
96    ResponseEntity<DomainGroup> addGroup(
97        @ApiParam(value = "The domain group to add.") @Valid @RequestBody DomainGroup group);
98  
99    /**
100    * Get domain group by name.
101    *
102    * @param groupName the group name
103    * @return the domain group
104    */
105   @ApiOperation(
106       value = "Get a domain group by name.",
107       nickname = "getGroupByName",
108       response = DomainGroup.class,
109       tags = {"domain-group-management-controller"})
110   @ApiResponses(value = {
111       @ApiResponse(code = 200, message = "The domain group with the specified name.",
112           response = DomainGroup.class),
113       @ApiResponse(code = 400, message = "Bad request.",
114           response = RestApiException.class),
115       @ApiResponse(code = 404, message = "Not found",
116           response = RestApiException.class),
117       @ApiResponse(code = 500, message = "Fatal server error.",
118           response = RestApiException.class)
119   })
120   @RequestMapping(
121       value = "/api/groups/{groupName}",
122       produces = {"application/json"},
123       method = RequestMethod.GET)
124   ResponseEntity<DomainGroup> getGroupByName(
125       @ApiParam(value = "The domain group name.", required = true)
126       @PathVariable("groupName") String groupName);
127 
128   /**
129    * Update domain group.
130    *
131    * @param groupName   the group name
132    * @param domainGroup the domain group
133    * @return the updated domain group
134    */
135   @ApiOperation(
136       value = "Updates a domain group.",
137       nickname = "updateGroup",
138       response = DomainGroup.class,
139       tags = {"domain-group-management-controller"})
140   @ApiResponses(value = {
141       @ApiResponse(code = 200, message = "The updated domain group.", response = DomainGroup.class),
142       @ApiResponse(code = 400, message = "Bad request.",
143           response = RestApiException.class),
144       @ApiResponse(code = 404, message = "Group not found.",
145           response = RestApiException.class),
146       @ApiResponse(code = 500, message = "Fatal server error.",
147           response = RestApiException.class)
148   })
149   @RequestMapping(
150       value = "/api/groups/{groupName}",
151       produces = {"application/json"},
152       consumes = {"application/json"},
153       method = RequestMethod.PUT)
154   ResponseEntity<DomainGroup> updateGroup(
155       @ApiParam(value = "The name of the domain group.", required = true)
156       @PathVariable("groupName") String groupName,
157 
158       @ApiParam(value = "The domain group.", required = true)
159       @Valid @RequestBody DomainGroup domainGroup);
160 
161   /**
162    * Domain group exists.
163    *
164    * @param groupName the group name
165    * @return true if the group exists otherwise false
166    */
167   @ApiOperation(
168       value = "Checks whether a domain group exists.",
169       nickname = "groupExists",
170       response = Boolean.class,
171       tags = {"domain-group-management-controller"})
172   @ApiResponses(value = {
173       @ApiResponse(code = 200, message = "True if the group exists otherwise false.",
174           response = Boolean.class),
175       @ApiResponse(code = 500, message = "Fatal server error.",
176           response = RestApiException.class)
177   })
178   @RequestMapping(value = "/api/groups/{groupName}/exists",
179       produces = {"application/json"},
180       method = RequestMethod.GET)
181   ResponseEntity<Boolean> groupExists(
182       @ApiParam(value = "The name of the domain group.", required = true)
183       @PathVariable("groupName") String groupName);
184 
185   /**
186    * Delete domain group.
187    *
188    * @param groupName the group name
189    * @return {@code true} if the domain group was deleted, otherwise {@code false}
190    */
191   @ApiOperation(
192       value = "Delete domain group.",
193       nickname = "deleteGroup",
194       tags = {"domain-group-management-controller"})
195   @ApiResponses(value = {
196       @ApiResponse(code = 200,
197           message = "True if the domain group was successfully deleted, otherwise false.",
198           response = Boolean.class),
199       @ApiResponse(code = 400, message = "Bad request.",
200           response = RestApiException.class),
201       @ApiResponse(code = 500, message = "Fatal server error.",
202           response = RestApiException.class)
203   })
204   @RequestMapping(
205       value = "/api/groups/{groupName}",
206       produces = {"application/json"},
207       method = RequestMethod.DELETE)
208   ResponseEntity<Boolean> deleteGroup(
209       @ApiParam(value = "The domain group name.", required = true)
210       @PathVariable("groupName") String groupName);
211 
212 }
